区间加区间乘区间查询

区间加区间乘线段树


#include<cstdio>
#define ls rt<<1
#define rs rt<<1|1
const int N=100011;
int n,q,p;
typedef long long ll;
ll mul[N<<2],add[N<<2],s[N<<2];
void pu(int rt){s[rt]=(s[ls]+s[rs])%p;}
void pd(int rt,int ln,int rn){
    if(mul[rt]^1){
        (mul[ls]*=mul[rt])%=p;
        (mul[rs]*=mul[rt])%=p;
        (add[ls]*=mul[rt])%=p;
        (add[rs]*=mul[rt])%=p;
        (s[ls]*=mul[rt])%=p;
        (s[rs]*=mul[rt])%=p;
        mul[rt]=1;
    }
    if(add[rt]){
        (add[ls]+=add[rt])%=p;
        (add[rs]+=add[rt])%=p;
        (s[ls]+=add[rt]*ln%p)%=p;
        (s[rs]+=add[rt]*rn%p)%=p;
        add[rt]=0;
    }
}
void build(int l,int r,int rt){
    mul[rt]=1,add[rt]=0;
    if(l==r){scanf("%lld",s+rt);return;}
    int m=l+r>>1;
    build(l,m,ls),build(m+1,r,rs);
    pu(rt);
}
void upda(int L,int R,ll v,int l,int r,int rt){
    if(L<=l&&r<=R){
        (s[rt]+=v*(r-l+1)%p)%=p;
        (add[rt]+=v)%=p;
        return;
    }
    int m=l+r>>1;
    pd(rt,m-l+1,r-m);
    if(L<=m)upda(L,R,v,l,m,ls);
    if(R>m)upda(L,R,v,m+1,r,rs);
    pu(rt);
}
void updm(int L,int R,int v,int l,int r,int rt){
    if(L<=l&&r<=R){
        (s[rt]*=v)%=p,
        (add[rt]*=v)%=p,
        (mul[rt]*=v)%=p;
        return;
    }
    int m=l+r>>1;
    pd(rt,m-l+1,r-m);
    if(L<=m)updm(L,R,v,l,m,ls);
    if(R>m)updm(L,R,v,m+1,r,rs);
    pu(rt);
}
int ask(int L,int R,int l,int r,int rt){
    if(L<=l&&r<=R)return s[rt];
    int m=l+r>>1,ans=0;
    pd(rt,m-l+1,r-m);
    if(L<=m)ans=ask(L,R,l,m,ls);
    if(R>m)(ans+=ask(L,R,m+1,r,rs))%=p;
    return ans;
}
int main(){
    scanf("%d%d%d",&n,&q,&p);
    build(1,n,1);
    while(q--){
        int opt,l,r,v;
        scanf("%d%d%d",&opt,&l,&r);
        if(opt==1)scanf("%d",&v),updm(l,r,v,1,n,1);
        else if(opt==2)scanf("%d",&v),upda(l,r,v,1,n,1);
        else printf("%lld\n",ask(l,r,1,n,1));
    }
}
