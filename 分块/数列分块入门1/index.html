#include<bits/stdc++.h>
const int N=50001;
int n,a[N],bl[N],laz[233],blk;
void add(int l,int r,int c){
    if(bl[l]==bl[r])
        for(int i=l;i<=r;++i)a[i]+=c;
    else{
        for(int i=l;bl[i]==bl[l];++i)a[i]+=c;
        for(int i=bl[l]+1;i<bl[r];++i)laz[i]+=c;
        for(int i=r;bl[i]==bl[r];--i)a[i]+=c;
    }
}
int main(){
    scanf("%d",&n);
    blk=sqrt(n);
    for(int i=1;i<=n;++i)
        scanf("%d",a+i),
        bl[i]=(i-1)/blk+1;
    for(int t=1;t<=n;++t){
        int opt,l,r,c;
        scanf("%d%d%d%d",&opt,&l,&r,&c);
        if(opt==0)add(l,r,c);
        else printf("%d\n",a[r]+laz[bl[r]]);
    }
}
