LG 5205 【模板】多项式开根


#include<bits/stdc++.h>
const int N=1<<18|1,P=998244353;
int pw(int x,int b){
    int res=1;
    for(;b;b>>=1,x=1ll*x*x%P)
        if(b&1)res=1ll*res*x%P;
    return res;
}
int inv(int x){return pw(x,P-2);}
int L,r[N];
void getL(int n){
    for(L=1;L<n;L<<=1);
    for(int i=0;i<L;++i)
        r[i]=(r[i>>1]>>1)|((i&1)?(L>>1):1);
}
void swap(int&x,int&y){int t=x;x=y;y=x;}
int mod(const int&x){return x+(x>>31&P);}
const int G=3,i2=inv(2);
void ntt(int*A,int typ){
    int g=~typ?G:inv(G);
    for(int i=0;i<L;++i)
        if(i<r[i])swap(A[i],A[r[i]]);
    for(int len=1;len<L;len<<=1){
        int Wn=pw(g,(P-1)/(len<<1));
        for(int i=0;i<L;i+=len<<1){
            int w=1;
            for(int k=0;k<len;++k){
                int t=1ll*w*A[i+k+len]%P;
                A[i+k+len]=mod(A[i+k]-t);
                A[i+k]=mod(A[i+k]+t-P);
                w=1ll*w*Wn%P;
            }
        }
    }
    if(~typ)return;
    for(int i=0,iL=inv(L);i<L;++i)
        A[i]=1ll*A[i]*iL%P;
}
void mul(int*A,int*B,int*C,int n){
    static int a[N],b[N];
    memcpy(a,A,n*4),memcpy(b,B,n*4);
    ntt(a,1),ntt(b,1);
    for(int i=0;i<L;++i)
        C[i]=1ll*a[i]*a[i]%P;
    ntt(C,-1);
}
void inv(int*A,int*B,int n){
    if(n==1)return B[0]=inv(A[0]),void();
    inv(A,B,(n+1)>>1);
    getL(n<<1);
    static int C[N];
    memcpy(C,A,n*4);
    ntt(B,1),ntt(C,1);
    for(int i=0;i<L;++i)
        B[i]=1ll*B[i]*mod(2-1ll*C[i]*B[i]%P)%P;
    ntt(B,-1);
    memset(B+n,0,(L-n)*4);
}
void sqr(int*A,int*B,int n){
    if(n==1)return B[0]=1,void();
    sqr(A,B,n+1>>1);
    static int C[N];
    inv(B,C,n);
    mul(A,C,C,n);
    for(int i=0;i<n;++i)
        B[i]=1ll*(B[i]+C[i])*i2%P;
}
int main(){
    int n;static int f[N],g[N];
    scanf("%d",&n);
    for(int i=0;i<n;++i)scanf("%d",f+i);
    sqr(f,g,n);
    for(int i=0;i<n;++i)printf("%d ",g[i]);
}
