#include<bits/stdc++.h>
#define fl(i,x) for(int i=head[x],to;to=e[i].to,i;i=e[i].nxt)
void cmax(int&x,int y){if(y>x)x=y;}
const int N=50011;
int n,k,cnt,SZ,rt,mxs,head[N],siz[N],b[501];
long long ans,res;
bool v[N];
struct edge{int to,nxt;}e[N*2];
void add(int x,int y){e[++cnt].to=y;e[cnt].nxt=head[x];head[x]=cnt;}
void frt(int x,int f){
    siz[x]=1;
    int res=0;
    fl(i,x)if(!v[to]&&to!=f)
        frt(to,x),siz[x]+=siz[to],cmax(res,siz[to]);
    cmax(res,SZ-siz[x]);
    if(res<mxs)rt=x,mxs=res;
}
void gd(int x,int d,int f){
    if(d>k)return;
    res+=b[k-d];++b[d];
    fl(i,x)if(!v[to]&&to!=f)
        gd(to,d+1,x);
}
void calc(int x,int d){
    res=0,memset(b,0,sizeof b);
    gd(x,d,0);
}
void solve(int x){
    v[x]=1;
    calc(x,0),ans+=res;
    fl(i,x)if(!v[to]){
        calc(to,1),ans-=res;
        rt=0,mxs=-1u>>1,SZ=siz[to];
        frt(to,0);
        solve(rt);
    }
}
int main(){
    scanf("%d%d",&n,&k);
    int x,y;
    for(int i=1;i<n;++i)   
        scanf("%d%d",&x,&y),
        add(x,y),add(y,x);
    rt=0,mxs=-1u>>1,SZ=n;
    frt(1,0);
    solve(rt);
    printf("%lld\n",ans);
}
