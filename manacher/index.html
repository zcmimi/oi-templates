LG 3805 【模板】manacher算法


#include<bits/stdc++.h>
const int N=11000001;
int n,p[N*2];
char a[N],s[N*2];
int min(int x,int y){return x<y?x:y;}
int main(){
    scanf("%s",a+1);
    n=strlen(a+1)<<1|1;
    for(int i=1;i<=n;++i)
        s[i]=(i&1)?'#':a[i>>1];
    int R=0,m=0;
    for(int i=1;i<=n;++i){
        if(i<=R)p[i]=min(R-i,p[m*2-i]);
        while(i+p[i]+1<=n&&i-p[i]-1>=1&&s[i+p[i]+1]==s[i-p[i]-1])++p[i];
        if(i+p[i]>R)R=i+p[i],m=i;
    }
    int ans=0;
    for(int i=1;i<=n;++i)
        if(p[i]>ans)ans=p[i];
    printf("%d\n",ans);
}
