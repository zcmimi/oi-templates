#include<bits/stdc++.h>
const int N=50001;
typedef long long ll;
int n,blk,a[N],laz[233],bl[N];
ll s[233];
void add(int l,int r,int c){
    if(bl[l]==bl[r]){
        for(int i=l;i<=r;++i)
            a[i]+=c,s[bl[i]]+=c;
        return;
    }
    for(int i=l;bl[i]==bl[l];++i)
        a[i]+=c,s[bl[i]]+=c;
    for(int i=r;bl[i]==bl[r];--i)
        a[i]+=c,s[bl[i]]+=c;
    for(int i=bl[l]+1;i<bl[r];++i)
        laz[i]+=c,s[i]+=1ll*c*blk;
}
int qry(int l,int r,int p){
    ll res=0;
    if(bl[l]==bl[r]){
        for(int i=l;i<=r;++i)
            (res+=a[i]+laz[bl[i]])%=p;
        return res;
    }
    for(int i=l;bl[i]==bl[l];++i)
        (res+=a[i]+laz[bl[i]])%=p;
    for(int i=r;bl[i]==bl[r];--i)
        (res+=a[i]+laz[bl[i]])%=p;
    for(int i=bl[l]+1;i<bl[r];++i)
        (res+=s[i])%=p;
    return res;
}
int main(){
    scanf("%d",&n);
    blk=sqrt(n);
    for(int i=1;i<=n;++i)
        scanf("%d",a+i),
        bl[i]=(i-1)/blk+1,
        s[bl[i]]+=a[i];
    for(int t=1;t<=n;++t){
        int opt,l,r,c;
        scanf("%d%d%d%d",&opt,&l,&r,&c);
        if(opt==0)add(l,r,c);
        else printf("%d\n",qry(l,r,c+1));
    }
}
