区间加区间查询

LG P3372 【模板】线段树 1


#include<cstdio>
const int N=100011;
int n,q;
typedef long long ll;
ll s[N],S[N];
void upd(int x,int v){
    for(int i=x;i<=n;i+=i&-i)s[i]+=v,S[i]+=1ll*(x-1)*v;
}
ll ask(int x){
    ll res=0;
    for(int i=x;i;i-=i&-i)
        res+=s[i]*x-S[i];
    return res;
}
int main(){
    scanf("%d%d",&n,&q);
    int opt,l,r,v;
    for(int i=1;i<=n;++i)scanf("%d",&v),upd(i,v),upd(i+1,-v);
    while(q--){
        scanf("%d%d%d",&opt,&l,&r,&v);
        if(opt==1)scanf("%d",&v),upd(l,v),upd(r+1,-v);
        else printf("%lld\n",ask(r)-ask(l-1));
    }
}
