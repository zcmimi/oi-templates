单调队列

LG P1886 滑动窗口 /【模板】单调队列


#include<bits/stdc++.h>
const int N=1000001;
int n,k,a[N],q[N];
int main(){
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n;++i)scanf("%d",a+i);
    int h=1,t=0;
    for(int i=1;i<=n;++i){
        while(h<=t&&a[i]<=a[q[t]])--t;
        q[++t]=i;
        if(i<k)continue;
        while(i-q[h]+1>k)++h;
        printf("%d ",a[q[h]]);
    }
    putchar('\n');
    h=1,t=0;
    for(int i=1;i<=n;++i){
        while(h<=t&&a[i]>=a[q[t]])--t;
        q[++t]=i;
        if(i<k)continue;
        while(i-q[h]+1>k)++h;
        printf("%d ",a[q[h]]);
    }
}
